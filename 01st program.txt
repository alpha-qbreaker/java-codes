import java.io.*;

class GFG {
    // Function to calculate the following series
    static int Series(int n) {
        int i;
        int sums = 0;
        for (i = 1; i <= n; i++) 
            sums += 1 / (i * i); // This will still use integer division
        return sums;
    }

    // Driver Code
    public static void main(String[] args) {
        int n = 3;
        int res = Series(n);
        System.out.println(res);
    }
}
